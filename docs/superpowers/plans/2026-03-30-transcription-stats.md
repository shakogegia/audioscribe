# Transcription Stats Page Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a `/stats` page with transcription analytics — overview cards, processing breakdown, transcription speed, model usage, pipeline health, and activity timeline.

**Architecture:** Server component page that calls a server action to query all stats from SQLite (Prisma). Chart components use shadcn/ui `chart` (Recharts wrapper). Book titles fetched via Audiobookshelf batch API. One small Python worker change to store `compute_type` in job metadata.

**Tech Stack:** Next.js App Router, shadcn/ui chart + card + table, Recharts (via shadcn), Prisma, Audiobookshelf API

### Task 4: Server action — getTranscriptionStats()

**Files:**

- [ ] **Step 1: Create the server action file**

Create `src/app/(authenticated)/(navigation)/stats/actions.ts` with the full stats query logic. This is the data layer — all computation happens here.

```typescript
"use server"

import { prisma } from "@/lib/prisma"
import { getBatchLibraryItems } from "@/lib/audiobookshelf"

export interface StatsData {
  overview: {
    booksTranscribed: number
    audioHoursProcessed: number
    medianRtf: number
    successRate: number
  }
  processingBreakdown: Array<{
    bookId: string
    bookTitle: string
    download: number
    prepare: number
    transcribe: number
  }>
  transcriptionSpeed: Array<{
    bookId: string
    bookTitle: string
    rtf: number
    model: string
    audioDuration: number
    processingTime: number
  }>
  modelUsage: Array<{
    model: string
    bookCount: number
    avgRtf: number
    medianRtf: number
    totalHours: number
  }>
  pipelineHealth: {
    failureRates: Array<{ stage: string; rate: number; total: number; failed: number }>
    retryDistribution: Array<{ attempts: number; count: number }>
    recentErrors: Array<{
      bookId: string
      bookTitle: string
      stage: string
      error: string
      timestamp: string
    }>
  }
  activity: Array<{
    period: string
    booksProcessed: number
    audioHours: number
  }>
}

function median(values: number[]): number {
  if (values.length === 0) return 0
  const sorted = [...values].sort((a, b) => a - b)
  const mid = Math.floor(sorted.length / 2)
  return sorted.length % 2 !== 0 ? sorted[mid] : (sorted[mid - 1] + sorted[mid]) / 2
}

function jobDurationMinutes(job: { startedAt: Date | null; completedAt: Date | null }): number {
  if (!job.startedAt || !job.completedAt) return 0
  return (job.completedAt.getTime() - job.startedAt.getTime()) / 60000
}

export async function getTranscriptionStats(): Promise<StatsData> {
  const books = await prisma.book.findMany({
    where: { transcribed: true },
    include: {
      jobs: true,
      audioChunks: true,
    },
    orderBy: { createdAt: "desc" },
    take: 50,
  })

  // Fetch titles from Audiobookshelf
  const bookIds = books.map(b => b.id)
  let titleMap = new Map<string, string>()
  try {
    const libraryItems = await getBatchLibraryItems(bookIds)
    titleMap = new Map(libraryItems.map(item => [item.id, item.title]))
  } catch {
    // Audiobookshelf unavailable — fall back to IDs
  }
  const getTitle = (id: string) => titleMap.get(id) || id.slice(0, 8)

  // All jobs (including non-transcribed books) for pipeline health
  const allJobs = await prisma.job.findMany()

  // --- Overview ---
  const totalAudioSeconds = books.reduce(
    (sum, b) => sum + b.audioChunks.reduce((s, c) => s + c.duration, 0),
    0
  )
  const audioHoursProcessed = totalAudioSeconds / 3600

  const transcribeJobs = allJobs.filter(j => j.type === "Transcribe")
  const completedTranscribe = transcribeJobs.filter(j => j.status === "Completed")
  const failedTranscribe = transcribeJobs.filter(j => j.status === "Failed")
  const successRate =
    completedTranscribe.length + failedTranscribe.length > 0
      ? (completedTranscribe.length / (completedTranscribe.length + failedTranscribe.length)) * 100
      : 100

  // Per-book RTF for median calculation
  const bookRtfs: number[] = []

  // --- Processing Breakdown & Transcription Speed ---
  const processingBreakdown: StatsData["processingBreakdown"] = []
  const transcriptionSpeed: StatsData["transcriptionSpeed"] = []

  for (const book of books) {
    const audioDurationMin = book.audioChunks.reduce((s, c) => s + c.duration, 0) / 60
    if (audioDurationMin === 0) continue

    const downloadJobs = book.jobs.filter(j => j.type === "Download" && j.status === "Completed")
    const prepareJobs = book.jobs.filter(j => j.type === "PrepareAudio" && j.status === "Completed")
    const transcribeJobsForBook = book.jobs.filter(j => j.type === "Transcribe" && j.status === "Completed")

    const downloadMin = downloadJobs.reduce((s, j) => s + jobDurationMinutes(j), 0)
    const prepareMin = prepareJobs.reduce((s, j) => s + jobDurationMinutes(j), 0)
    const transcribeMin = transcribeJobsForBook.reduce((s, j) => s + jobDurationMinutes(j), 0)

    processingBreakdown.push({
      bookId: book.id,
      bookTitle: getTitle(book.id),
      download: Math.round(downloadMin * 10) / 10,
      prepare: Math.round(prepareMin * 10) / 10,
      transcribe: Math.round(transcribeMin * 10) / 10,
    })

    if (transcribeMin > 0) {
      const rtf = audioDurationMin / transcribeMin
      bookRtfs.push(rtf)

      // Get model from first transcribe job metadata
      let model = "unknown"
      const firstJob = transcribeJobsForBook[0]
      if (firstJob?.metadata) {
        try {
          model = JSON.parse(firstJob.metadata).model || "unknown"
        } catch {}
      }

      transcriptionSpeed.push({
        bookId: book.id,
        bookTitle: getTitle(book.id),
        rtf: Math.round(rtf * 100) / 100,
        model,
        audioDuration: Math.round(audioDurationMin * 10) / 10,
        processingTime: Math.round(transcribeMin * 10) / 10,
      })
    }
  }

  transcriptionSpeed.sort((a, b) => b.rtf - a.rtf)

  // --- Model Usage ---
  const modelMap = new Map<string, { rtfs: number[]; totalHours: number; bookIds: Set<string> }>()
  for (const entry of transcriptionSpeed) {
    const existing = modelMap.get(entry.model) || { rtfs: [], totalHours: 0, bookIds: new Set() }
    existing.rtfs.push(entry.rtf)
    existing.totalHours += entry.audioDuration / 60
    existing.bookIds.add(entry.bookId)
    modelMap.set(entry.model, existing)
  }

  const modelUsage: StatsData["modelUsage"] = Array.from(modelMap.entries()).map(
    ([model, data]) => ({
      model,
      bookCount: data.bookIds.size,
      avgRtf: Math.round((data.rtfs.reduce((s, v) => s + v, 0) / data.rtfs.length) * 100) / 100,
      medianRtf: Math.round(median(data.rtfs) * 100) / 100,
      totalHours: Math.round(data.totalHours * 10) / 10,
    })
  )

  // --- Pipeline Health ---
  const stages = ["Download", "PrepareAudio", "Transcribe"] as const
  const failureRates = stages.map(stage => {
    const stageJobs = allJobs.filter(j => j.type === stage)
    const failed = stageJobs.filter(j => j.status === "Failed").length
    return {
      stage,
      rate: stageJobs.length > 0 ? Math.round((failed / stageJobs.length) * 1000) / 10 : 0,
      total: stageJobs.length,
      failed,
    }
  })

  const retryMap = new Map<number, number>()
  for (const job of allJobs) {
    if (job.attempts > 1) {
      retryMap.set(job.attempts, (retryMap.get(job.attempts) || 0) + 1)
    }
  }
  const retryDistribution = Array.from(retryMap.entries())
    .map(([attempts, count]) => ({ attempts, count }))
    .sort((a, b) => a.attempts - b.attempts)

  const recentErrors = allJobs
    .filter(j => j.status === "Failed" && j.error)
    .sort((a, b) => (b.completedAt?.getTime() || 0) - (a.completedAt?.getTime() || 0))
    .slice(0, 10)
    .map(j => ({
      bookId: j.bookId,
      bookTitle: getTitle(j.bookId),
      stage: j.type,
      error: j.error!,
      timestamp: (j.completedAt || j.updatedAt).toISOString(),
    }))

  // --- Activity Over Time ---
  const activityMap = new Map<string, { books: Set<string>; audioSeconds: number }>()
  for (const book of books) {
    const lastJob = book.jobs
      .filter(j => j.status === "Completed")
      .sort((a, b) => (b.completedAt?.getTime() || 0) - (a.completedAt?.getTime() || 0))[0]
    if (!lastJob?.completedAt) continue

    const period = lastJob.completedAt.toISOString().slice(0, 7) // "YYYY-MM"
    const existing = activityMap.get(period) || { books: new Set(), audioSeconds: 0 }
    existing.books.add(book.id)
    existing.audioSeconds += book.audioChunks.reduce((s, c) => s + c.duration, 0)
    activityMap.set(period, existing)
  }

  const activity = Array.from(activityMap.entries())
    .map(([period, data]) => ({
      period,
      booksProcessed: data.books.size,
      audioHours: Math.round((data.audioSeconds / 3600) * 10) / 10,
    }))
    .sort((a, b) => a.period.localeCompare(b.period))

  return {
    overview: {
      booksTranscribed: books.length,
      audioHoursProcessed: Math.round(audioHoursProcessed * 10) / 10,
      medianRtf: Math.round(median(bookRtfs) * 100) / 100,
      successRate: Math.round(successRate * 10) / 10,
    },
    processingBreakdown,
    transcriptionSpeed,
    modelUsage,
    pipelineHealth: { failureRates, retryDistribution, recentErrors },
    activity,
  }
}
```

```typescript
"use client"

import { Card, CardContent, CardDescription, CardHeader, CardTitle } from "@/components/ui/card"
import { ChartConfig, ChartContainer, ChartTooltip, ChartTooltipContent } from "@/components/ui/chart"
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from "@/components/ui/table"
import { Badge } from "@/components/ui/badge"
import { Bar, BarChart, CartesianGrid, XAxis, YAxis } from "recharts"
import type { StatsData } from "../actions"

const chartConfig = {
  rate: { label: "Failure Rate", color: "var(--chart-1)" },
} satisfies ChartConfig

export function PipelineHealth({ data }: { data: StatsData["pipelineHealth"] }) {
  const hasFailures = data.failureRates.some(f => f.failed > 0)
  const hasRetries = data.retryDistribution.length > 0
  const hasErrors = data.recentErrors.length > 0

  if (!hasFailures && !hasRetries && !hasErrors) {
    return (
      <Card>
        <CardHeader>
          <CardTitle>Pipeline Health</CardTitle>
          <CardDescription>No failures or retries recorded</CardDescription>
        </CardHeader>
        <CardContent>
          <p className="text-sm text-muted-foreground">All jobs completed successfully.</p>
        </CardContent>
      </Card>
    )
  }

  return (
    <div className="grid gap-4">
      <div className="grid gap-4 lg:grid-cols-2">
        {/* Failure Rate Chart */}
        <Card>
          <CardHeader>
            <CardTitle>Failure Rate by Stage</CardTitle>
            <CardDescription>Percentage of failed jobs per pipeline stage</CardDescription>
          </CardHeader>
          <CardContent>
            <ChartContainer config={chartConfig} className="min-h-[200px] w-full">
              <BarChart data={data.failureRates}>
                <CartesianGrid vertical={false} />
                <XAxis dataKey="stage" />
                <YAxis tickFormatter={v => `${v}%`} />
                <ChartTooltip
                  content={
                    <ChartTooltipContent
                      formatter={(value, _name, item) => {
                        const d = item.payload
                        return `${value}% (${d.failed}/${d.total} jobs)`
                      }}
                    />
                  }
                />
                <Bar dataKey="rate" fill="var(--color-rate)" radius={[4, 4, 0, 0]} />
              </BarChart>
            </ChartContainer>
          </CardContent>
        </Card>

        {/* Retry Distribution */}
        <Card>
          <CardHeader>
            <CardTitle>Retries</CardTitle>
            <CardDescription>Jobs that required retry attempts</CardDescription>
          </CardHeader>
          <CardContent>
            {hasRetries ? (
              <div className="space-y-2">
                {data.retryDistribution.map(entry => (
                  <div key={entry.attempts} className="flex items-center justify-between">
                    <span className="text-sm">
                      {entry.attempts} attempt{entry.attempts > 1 ? "s" : ""}
                    </span>
                    <Badge variant="secondary">{entry.count} jobs</Badge>
                  </div>
                ))}
              </div>
            ) : (
              <p className="text-sm text-muted-foreground">No retries needed.</p>
            )}
          </CardContent>
        </Card>
      </div>

      {/* Recent Errors */}
      {hasErrors && (
        <Card>
          <CardHeader>
            <CardTitle>Recent Errors</CardTitle>
            <CardDescription>Last {data.recentErrors.length} failed jobs</CardDescription>
          </CardHeader>
          <CardContent>
            <Table>
              <TableHeader>
                <TableRow>
                  <TableHead>Book</TableHead>
                  <TableHead>Stage</TableHead>
                  <TableHead>Error</TableHead>
                  <TableHead className="text-right">Time</TableHead>
                </TableRow>
              </TableHeader>
              <TableBody>
                {data.recentErrors.map((err, i) => (
                  <TableRow key={i}>
                    <TableCell className="font-medium max-w-[150px] truncate">
                      {err.bookTitle}
                    </TableCell>
                    <TableCell>
                      <Badge variant="outline">{err.stage}</Badge>
                    </TableCell>
                    <TableCell className="max-w-[300px] truncate text-sm text-muted-foreground">
                      {err.error}
                    </TableCell>
                    <TableCell className="text-right text-sm text-muted-foreground whitespace-nowrap">
                      {new Date(err.timestamp).toLocaleDateString()}
                    </TableCell>
                  </TableRow>
                ))}
              </TableBody>
            </Table>
          </CardContent>
        </Card>
      )}
    </div>
  )
}
```
